\chapter{特征值与不变子空间}

本章进入算子理论的核心。
与传统路线不同，我们\textbf{不使用特征多项式，也不使用行列式}。
特征值的存在、上三角化、可对角化等结论，
都将从不变子空间与多项式作用于算子的观点出发得到。

设 $V$ 是有限维向量空间，$T\in\End(V)$。
如果没有特别说明，本章默认 $V\neq\{0\}$。

\section{不变子空间}

\begin{definition}
子空间 $U\subseteq V$ 称为 $T$-\emph{不变}，
如果对每个 $u\in U$ 都有 $Tu\in U$。
等价地，$T(U)\subseteq U$。
\end{definition}

\begin{example}
零子空间 $\{0\}$ 与整个空间 $V$ 总是 $T$-不变的。
\end{example}

\begin{example}
若 $v\neq 0$ 且 $Tv=\lambda v$，
则一维子空间 $\spn(v)$ 是 $T$-不变的。
反过来，若 $\spn(v)$ 是 $T$-不变的，
则必有某个标量 $\lambda$ 使 $Tv=\lambda v$。
因此一维不变子空间正是特征向量张成的子空间。
\end{example}

\begin{proposition}
若 $U_1,U_2\subseteq V$ 都是 $T$-不变子空间，
则 $U_1+U_2$ 与 $U_1\cap U_2$ 也是 $T$-不变的。
\end{proposition}

\begin{proof}
若 $u_1\in U_1,u_2\in U_2$，则
\[
T(u_1+u_2)=Tu_1+Tu_2\in U_1+U_2,
\]
所以 $U_1+U_2$ 不变。
若 $u\in U_1\cap U_2$，则 $Tu\in U_1$ 且 $Tu\in U_2$，
故 $Tu\in U_1\cap U_2$。
\end{proof}

\begin{proposition}
设 $U$ 是 $T$-不变子空间。
则限制映射
\[
T|_U:U\to U
\]
是 $U$ 上的线性算子。
\end{proposition}

\begin{proof}
由不变性，$T(u)\in U$ 对每个 $u\in U$ 成立，
因此 $T|_U$ 的值域确在 $U$ 中。
线性性由 $T$ 的线性性继承。
\end{proof}

\section{特征值与特征向量}

\begin{definition}
标量 $\lambda\in\F$ 称为 $T$ 的\emph{特征值}，
如果算子 $T-\lambda\Id$ 不可逆。
非零向量 $v\in V$ 称为对应于 $\lambda$ 的\emph{特征向量}，
如果
\[
Tv=\lambda v.
\]
\end{definition}

\begin{proposition}
对 $\lambda\in\F$，下列条件等价：
\begin{enumerate}[label=(\alph*)]
\item $\lambda$ 是 $T$ 的特征值；
\item 存在非零向量 $v$ 使 $Tv=\lambda v$；
\item $\nul(T-\lambda\Id)\neq\{0\}$；
\item $\range(T-\lambda\Id)\neq V$。
\end{enumerate}
\end{proposition}

\begin{proof}
(a)$\Leftrightarrow$(c)$\Leftrightarrow$(d) 来自有限维空间上线性算子“可逆、单射、满射”等价。
而 (b) 与 (c) 只是把
\[
(T-\lambda\Id)v=0
\]
改写为 $Tv=\lambda v$。
\end{proof}

\begin{definition}
对应于特征值 $\lambda$ 的\emph{特征空间}定义为
\[
E(\lambda,T)=\nul(T-\lambda\Id).
\]
\end{definition}

\begin{proposition}
$E(\lambda,T)$ 是 $V$ 的子空间；
当且仅当 $\lambda$ 是特征值时，它非零。
\end{proposition}

\begin{proof}
它是线性映射 $T-\lambda\Id$ 的核，因此是子空间。
由定义，$\lambda$ 是特征值恰当且仅当这个核非零。
\end{proof}

\begin{lemma}
属于不同特征值的特征向量线性无关。
\end{lemma}

\begin{proof}
设 $v_1,\dots,v_m$ 分别对应互异特征值 $\lambda_1,\dots,\lambda_m$。
对 $m$ 归纳。
若
\[
a_1v_1+\cdots+a_mv_m=0,
\]
施以 $T-\lambda_m\Id$ 得
\[
a_1(\lambda_1-\lambda_m)v_1+\cdots+a_{m-1}(\lambda_{m-1}-\lambda_m)v_{m-1}=0.
\]
由归纳假设，$a_1=\cdots=a_{m-1}=0$，再回原式得 $a_m=0$。
\end{proof}

\begin{corollary}
若 $\lambda_1,\dots,\lambda_m$ 互异，
则特征空间之和
\[
E(\lambda_1,T)+\cdots+E(\lambda_m,T)
\]
是直和。
\end{corollary}

\begin{proof}
从每个特征空间取一组基，这些基向量都对应互异特征值，
故总列表线性无关。
\end{proof}

\section{复向量空间上的特征值存在性}

这一节是 LADR 风格的关键转折点。
我们先证明某个非零多项式一定会把算子“杀掉”。

\begin{lemma}
设 $V$ 为有限维向量空间，$\dim V=n$，$T\in\End(V)$。
则存在非零多项式 $p\in\F[x]$，使得
\[
p(T)=0.
\]
\end{lemma}

\begin{proof}
空间 $\End(V)$ 的维数为 $n^2$。
因此 $\Id,T,T^2,\dots,T^{n^2}$ 这 $n^2+1$ 个向量在 $\End(V)$ 中线性相关。
故存在不全为零的标量 $a_0,\dots,a_{n^2}$ 使
\[
a_0\Id+a_1T+\cdots+a_{n^2}T^{n^2}=0.
\]
令
\[
p(x)=a_0+a_1x+\cdots+a_{n^2}x^{n^2},
\]
则 $p\neq 0$ 且 $p(T)=0$。
\end{proof}

\begin{theorem}
设 $V$ 是非零有限维复向量空间，$T\in\End(V)$。
则 $T$ 至少有一个特征值。
\end{theorem}

\begin{proof}
由上引理，存在非零 $p\in\C[x]$ 使 $p(T)=0$。
由代数基本定理，
\[
p(x)=c(x-\lambda_1)\cdots(x-\lambda_m)
\]
其中 $c\neq 0$。
于是
\[
0=p(T)=c(T-\lambda_1\Id)\cdots(T-\lambda_m\Id).
\]
若每个 $T-\lambda_j\Id$ 都可逆，则其乘积可逆，不可能等于零算子。
故至少有一个 $T-\lambda_j\Id$ 不可逆。
所以某个 $\lambda_j$ 是特征值。
\end{proof}

\begin{remark}
请注意：这里没有使用特征多项式。
特征值的存在性来自“某个多项式在 $T$ 上为零”以及复系数多项式可分解为一次因子。
\end{remark}

\section{上三角矩阵}

\begin{theorem}
设 $V$ 是非零有限维复向量空间，$T\in\End(V)$。
则 $V$ 存在一组基，使得 $T$ 在这组基下的矩阵是上三角矩阵。
\end{theorem}

\begin{proof}
对 $n=\dim V$ 归纳。
当 $n=1$ 时显然。
设 $n>1$ 且结论对维数 $n-1$ 成立。
由上一节，$T$ 有特征值 $\lambda$，取对应特征向量 $v_1\neq 0$。
于是 $U=\spn(v_1)$ 是一维 $T$-不变子空间。

考虑商空间 $V/U$ 上诱导算子
\[
\widetilde T(v+U)=Tv+U.
\]
它定义良好，因为 $U$ 不变。
$V/U$ 的维数为 $n-1$，故由归纳假设，存在 $V/U$ 的一组基
\[
v_2+U,\dots,v_n+U
\]
使 $\widetilde T$ 在该基下为上三角。
于是
\[
Tv_k\in \spn(v_1,\dots,v_k)
\quad (k=2,\dots,n),
\]
而 $Tv_1=\lambda v_1$。
因此在基 $v_1,\dots,v_n$ 下，$T$ 的矩阵为上三角。
\end{proof}

\begin{corollary}
设 $V$ 为复向量空间，$T\in\End(V)$。
则 $T$ 的对角线上元素可取为 $T$ 的全部特征值（按重复出现计）。
\end{corollary}

\begin{proof}
在上三角基下，若对角线上第 $k$ 个元素为 $a_{kk}$，
则
\[
(T-a_{kk}\Id)v_k\in \spn(v_1,\dots,v_{k-1}),
\]
特别地，限制到商空间 $\spn(v_1,\dots,v_k)/\spn(v_1,\dots,v_{k-1})$ 上，
它作用为零，故这些对角元正是逐层得到的特征值。
反过来，上三角矩阵的任何特征值都必须出现在对角线上，
因为若 $T-\lambda\Id$ 可逆，则其上三角矩阵对角线都非零；
而若某个对角元为零，则它不可能可逆。
\end{proof}

\section{对角化}

\begin{definition}
若 $V$ 存在一组由 $T$ 的特征向量组成的基，
则称 $T$ \emph{可对角化}。
等价地，$T$ 在某组基下有对角矩阵表示。
\end{definition}

\begin{theorem}
设 $\lambda_1,\dots,\lambda_m$ 是 $T$ 的互异特征值。
则下列条件等价：
\begin{enumerate}[label=(\alph*)]
\item $T$ 可对角化；
\item $V=E(\lambda_1,T)\oplus\cdots\oplus E(\lambda_m,T)$；
\item 存在由特征向量组成的基。
\end{enumerate}
\end{theorem}

\begin{proof}
(a)$\Leftrightarrow$(c) 只是定义改写。
(c)$\Rightarrow$(b)：把基按特征值分组即可；不同特征值对应的特征空间和为直和。
(b)$\Rightarrow$(c)：分别取各特征空间的基，再合并，得到由特征向量组成的 $V$ 的基。
\end{proof}

\begin{corollary}
若 $T$ 在 $n$ 维空间上有 $n$ 个互异特征值，则 $T$ 可对角化。
\end{corollary}

\begin{proof}
每个特征值至少给出一个非零特征向量，互异特征值对应向量线性无关，
故得到 $n$ 个线性无关向量，形成基。
\end{proof}

\begin{proposition}
若 $T$ 可对角化，则对每个 $v\in V$，
可以唯一地写成若干特征向量的和。
\end{proposition}

\begin{proof}
由前述定理，$V$ 是特征空间的直和。
直和分解的向量表达自动唯一。
\end{proof}

\section{不变子空间与限制算子}

\begin{proposition}
若 $U$ 是 $T$-不变子空间，则 $T|_U$ 的每个特征值都是 $T$ 的特征值。
\end{proposition}

\begin{proof}
若 $\lambda$ 是 $T|_U$ 的特征值，取非零 $u\in U$ 使
\[
T|_U(u)=\lambda u.
\]
但 $T|_U(u)=Tu$，所以 $Tu=\lambda u$，
故 $u$ 也是 $T$ 的特征向量。
\end{proof}

\begin{proposition}
若 $U$ 是 $T$-不变子空间，则每个特征空间与 $U$ 的交
\[
E(\lambda,T)\cap U
\]
恰为 $E(\lambda,T|_U)$。
\end{proposition}

\begin{proof}
向量 $u\in U$ 属于 $E(\lambda,T|_U)$ 当且仅当
\[
T|_U(u)=\lambda u,
\]
这等价于 $Tu=\lambda u$ 且 $u\in U$，
也即 $u\in E(\lambda,T)\cap U$。
\end{proof}

\begin{remark}
本章的很多论证都可概括为：
先找到一个非平凡不变子空间，再把问题限制到它或投射到商空间上，
然后做归纳。
这是“算子优先”观点的典型思路。
\end{remark}
